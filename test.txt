// 导入Scanner类，用于接收用户输入
import java.util.Scanner;

public class PrimeCheck {
    public static void main(String[] args) {
        // 创建Scanner对象
        Scanner scanner = new Scanner(System.in);

        // 提示用户输入数字
        System.out.print("请输入一个整数：");
        int num = scanner.nextInt();

        // 调用判断质数的方法，接收结果
        boolean isPrime = isPrimeNumber(num);

        // 输出结果
        if (isPrime) {
            System.out.println(num + " 是质数");
        } else {
            System.out.println(num + " 不是质数");
        }

        // 关闭Scanner
        scanner.close();
    }

    /**
     * 判断一个数是否为质数
     * @param num 待判断的数字
     * @return 是质数返回true，否则返回false
     */
    public static boolean isPrimeNumber(int num) {
        // 1. 小于等于1的数，绝对不是质数
        if (num <= 1) {
            return false;
        }

        // 2. 2是最小的质数，也是唯一的偶数质数
        if (num == 2) {
            return true;
        }

        // 3. 所有大于2的偶数，都不是质数
        if (num % 2 == 0) {
            return false;
        }

        // 4. 检查从3到num的平方根的奇数（优化：无需遍历到num-1）
        // 原理：若num有大于平方根的因数，必然有一个小于平方根的对应因数
        for (int i = 3; i <= Math.sqrt(num); i += 2) {
            if (num % i == 0) {
                return false;
            }
        }

        // 所有条件都满足，是质数
        return true;
    }
}
